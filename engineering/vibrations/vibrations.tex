\section*{Single Degree of Freedom (SDOF) Systems}

\subsection*{Modeling and Equations of Motion}

\textbf{Translational System:}
$$m\ddot{x} + c\dot{x} + kx = f(t)$$

\textbf{Rotational System:}
$$I\ddot{\theta} + c_\theta\dot{\theta} + k_\theta\theta = M(t)$$

\textbf{Standard Form:}
$$\ddot{x} + 2\zeta\omega_n\dot{x} + \omega_n^2 x = \frac{f(t)}{m}$$

\subsection*{System Parameters}

\textbf{Natural Frequency:}
$$\omega_n = \sqrt{\frac{k}{m}} = 2\pi f_n \quad \text{(rad/s)}$$

\textbf{Natural Period:}
$$T_n = \frac{2\pi}{\omega_n} = \frac{1}{f_n} \quad \text{(s)}$$

\textbf{Damping Ratio:}
$$\zeta = \frac{c}{c_c} = \frac{c}{2\sqrt{km}} = \frac{c}{2m\omega_n}$$

\textbf{Critical Damping Coefficient:}
$$c_c = 2\sqrt{km} = 2m\omega_n$$

\textbf{Damped Natural Frequency:}
$$\omega_d = \omega_n\sqrt{1 - \zeta^2} \quad (\zeta < 1)$$

\subsection*{Equivalent Springs}

\textbf{Series:}
$$\frac{1}{k_{eq}} = \frac{1}{k_1} + \frac{1}{k_2} + \cdots$$

\textbf{Parallel:}
$$k_{eq} = k_1 + k_2 + \cdots$$

\subsection*{Equivalent Dampers}

\textbf{Series:}
$$\frac{1}{c_{eq}} = \frac{1}{c_1} + \frac{1}{c_2} + \cdots$$

\textbf{Parallel:}
$$c_{eq} = c_1 + c_2 + \cdots$$

\section*{Free Vibration}

\subsection*{Undamped Free Vibration ($c = 0$)}

\textbf{Equation of Motion:}
$$m\ddot{x} + kx = 0$$

\textbf{General Solution:}
$$x(t) = A\cos(\omega_n t) + B\sin(\omega_n t)$$

Or equivalently:
$$x(t) = C\cos(\omega_n t - \phi)$$

where $C = \sqrt{A^2 + B^2}$ and $\phi = \tan^{-1}(B/A)$

\textbf{Using Initial Conditions:}

If $x(0) = x_0$ and $\dot{x}(0) = v_0$:
$$x(t) = x_0\cos(\omega_n t) + \frac{v_0}{\omega_n}\sin(\omega_n t)$$

\subsection*{Damped Free Vibration ($c \neq 0$)}

\textbf{Underdamped} ($\zeta < 1$):
$$x(t) = e^{-\zeta\omega_n t}(A\cos(\omega_d t) + B\sin(\omega_d t))$$

Or:
$$x(t) = Ce^{-\zeta\omega_n t}\cos(\omega_d t - \phi)$$

With initial conditions $x(0) = x_0$ and $\dot{x}(0) = v_0$:
$$x(t) = e^{-\zeta\omega_n t}\left[x_0\cos(\omega_d t) + \frac{v_0 + \zeta\omega_n x_0}{\omega_d}\sin(\omega_d t)\right]$$

\textbf{Critically Damped} ($\zeta = 1$):
$$x(t) = (A + Bt)e^{-\omega_n t}$$

With initial conditions:
$$x(t) = [x_0 + (v_0 + \omega_n x_0)t]e^{-\omega_n t}$$

\textbf{Overdamped} ($\zeta > 1$):
$$x(t) = Ae^{s_1 t} + Be^{s_2 t}$$

where $s_1, s_2 = -\zeta\omega_n \pm \omega_n\sqrt{\zeta^2 - 1}$

\subsection*{Logarithmic Decrement}

For underdamped systems, ratio of successive peaks:
$$\delta = \ln\left(\frac{x(t)}{x(t + T_d)}\right) = \frac{2\pi\zeta}{\sqrt{1 - \zeta^2}}$$

For $n$ cycles:
$$\delta = \frac{1}{n}\ln\left(\frac{x_0}{x_n}\right)$$

\textbf{Damping Ratio from Logarithmic Decrement:}
$$\zeta = \frac{\delta}{\sqrt{4\pi^2 + \delta^2}}$$

For small damping ($\zeta \ll 1$):
$$\zeta \approx \frac{\delta}{2\pi}$$

\section*{Harmonic Forced Vibration}

\subsection*{Harmonic Force Excitation}

\textbf{Equation of Motion:}
$$m\ddot{x} + c\dot{x} + kx = F_0\cos(\omega t)$$

or
$$\ddot{x} + 2\zeta\omega_n\dot{x} + \omega_n^2 x = \frac{F_0}{m}\cos(\omega t)$$

\textbf{Steady-State Response:}
$$x(t) = X\cos(\omega t - \phi)$$

\textbf{Amplitude:}
$$X = \frac{F_0/k}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}} = \frac{\delta_{st}}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}}$$

where:
- $r = \omega/\omega_n$ is the frequency ratio
- $\delta_{st} = F_0/k$ is the static deflection

\textbf{Magnification Factor:}
$$M(r) = \frac{X}{\delta_{st}} = \frac{1}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}}$$

\textbf{Phase Angle:}
$$\phi = \tan^{-1}\left(\frac{2\zeta r}{1 - r^2}\right)$$

\subsection*{Resonance}

\textbf{Peak Response Frequency:}
$$\omega_r = \omega_n\sqrt{1 - 2\zeta^2} \quad (\zeta < 1/\sqrt{2})$$

For small damping:
$$\omega_r \approx \omega_n$$

\textbf{Peak Amplitude at Resonance:}
$$X_{max} = \frac{F_0/k}{2\zeta\sqrt{1 - \zeta^2}} \approx \frac{F_0/k}{2\zeta} \quad (\zeta \ll 1)$$

\textbf{Quality Factor:}
$$Q = \frac{1}{2\zeta} = \frac{X_{max}}{\delta_{st}}$$

\subsection*{Special Frequency Ratios}

\textbf{At $r = 1$ (forced at natural frequency):}
$$X = \frac{F_0/k}{2\zeta}, \quad \phi = 90°$$

\textbf{At $r \ll 1$ (low frequency):}
$$X \approx \frac{F_0}{k}, \quad \phi \approx 0°$$

\textbf{At $r \gg 1$ (high frequency):}
$$X \approx \frac{F_0}{m\omega^2}, \quad \phi \approx 180°$$

\section*{Base Excitation}

\textbf{Motion of Base:}
$$y(t) = Y\sin(\omega t)$$

\textbf{Equation of Motion:}
$$m\ddot{x} + c(\dot{x} - \dot{y}) + k(x - y) = 0$$

\textbf{Relative Displacement:} $z = x - y$
$$m\ddot{z} + c\dot{z} + kz = -m\ddot{y} = m\omega^2 Y\sin(\omega t)$$

\textbf{Absolute Displacement Amplitude:}
$$X = Y\sqrt{\frac{1 + (2\zeta r)^2}{(1 - r^2)^2 + (2\zeta r)^2}}$$

\textbf{Displacement Transmissibility:}
$$T_d = \frac{X}{Y} = \sqrt{\frac{1 + (2\zeta r)^2}{(1 - r^2)^2 + (2\zeta r)^2}}$$

\section*{Force Transmissibility and Vibration Isolation}

\textbf{Force Transmitted to Base:}
$$F_T = \sqrt{(kX)^2 + (c\omega X)^2}$$

\textbf{Force Transmissibility:}
$$T_F = \frac{F_T}{F_0} = \sqrt{\frac{1 + (2\zeta r)^2}{(1 - r^2)^2 + (2\zeta r)^2}}$$

Note: $T_F = T_d$ (same formula)

\textbf{Isolation Region:}
$$T < 1 \quad \text{when} \quad r > \sqrt{2}$$

\textbf{For good isolation:}
\begin{itemize}
\item Choose $r > \sqrt{2}$ (typically $r > 3$ for practical systems)
\item Use small damping (but enough for transient response)
\item Lower the natural frequency (softer springs)
\end{itemize}

\textbf{Percent Isolation:}
$$\text{Isolation} = (1 - T) \times 100\%$$

\section*{Rotating Imbalance}

\textbf{Rotating Mass $m_0$ at Radius $e$ with Angular Velocity $\omega$:}

\textbf{Exciting Force:}
$$F(t) = m_0 e\omega^2\cos(\omega t)$$

\textbf{Steady-State Amplitude:}
$$X = \frac{m_0 e}{m}\frac{r^2}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}}$$

where $m$ is total mass and $r = \omega/\omega_n$

\textbf{Normalized Amplitude:}
$$\frac{mX}{m_0 e} = \frac{r^2}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}}$$

\textbf{Key Observations:}
\begin{itemize}
\item At low speeds ($r \ll 1$): $X \approx 0$
\item At resonance ($r = 1$): $X = \frac{m_0 e}{2m\zeta}$
\item At high speeds ($r \gg 1$): $X \approx \frac{m_0 e}{m}$ (independent of damping)
\end{itemize}

\section*{General Forcing Functions}

\subsection*{Impulse Response (Unit Impulse)}

\textbf{Impulse Response Function:}
$$h(t) = \frac{1}{m\omega_d}e^{-\zeta\omega_n t}\sin(\omega_d t) \quad (t \geq 0)$$

For undamped system:
$$h(t) = \frac{1}{m\omega_n}\sin(\omega_n t)$$

\subsection*{Convolution Integral (Duhamel's Integral)}

For arbitrary forcing $f(t)$:
$$x(t) = \int_0^t h(t - \tau)f(\tau)\,d\tau$$

$$x(t) = \frac{1}{m\omega_d}\int_0^t f(\tau)e^{-\zeta\omega_n(t-\tau)}\sin[\omega_d(t - \tau)]\,d\tau$$

\subsection*{Step Response}

For step input $f(t) = F_0 u(t)$:

\textbf{Undamped:}
$$x(t) = \frac{F_0}{k}[1 - \cos(\omega_n t)]$$

\textbf{Underdamped:}
$$x(t) = \frac{F_0}{k}\left[1 - e^{-\zeta\omega_n t}\left(\cos(\omega_d t) + \frac{\zeta}{\sqrt{1 - \zeta^2}}\sin(\omega_d t)\right)\right]$$

Or:
$$x(t) = \frac{F_0}{k}\left[1 - \frac{e^{-\zeta\omega_n t}}{\sqrt{1 - \zeta^2}}\cos(\omega_d t - \psi)\right]$$

where $\psi = \tan^{-1}\left(\frac{\zeta}{\sqrt{1 - \zeta^2}}\right)$

\subsection*{Ramp Response}

For ramp input $f(t) = F_0 \cdot t$:
$$x(t) = \frac{F_0}{k}\left[t - \frac{2\zeta}{\omega_n}\right] + \text{transient terms}$$

\section*{Frequency Response Methods}

\subsection*{Transfer Function}

$$H(s) = \frac{X(s)}{F(s)} = \frac{1}{ms^2 + cs + k} = \frac{1/m}{s^2 + 2\zeta\omega_n s + \omega_n^2}$$

\textbf{Frequency Response Function (FRF):}
$$H(j\omega) = H(s)|_{s=j\omega} = \frac{1/k}{(1 - r^2) + j(2\zeta r)}$$

\textbf{Magnitude:}
$$|H(j\omega)| = \frac{1/k}{\sqrt{(1 - r^2)^2 + (2\zeta r)^2}}$$

\textbf{Phase:}
$$\angle H(j\omega) = -\tan^{-1}\left(\frac{2\zeta r}{1 - r^2}\right)$$

\section*{Energy Methods}

\subsection*{Rayleigh's Method}

For finding natural frequency without solving EOM:

\textbf{Principle:} At maximum displacement, all energy is potential. At equilibrium, all energy is kinetic.
$$T_{max} = V_{max}$$

$$\frac{1}{2}m\omega_n^2 X^2 = \frac{1}{2}kX^2$$

$$\omega_n = \sqrt{\frac{k}{m}}$$

\textbf{For Distributed Systems:}
$$\omega_n = \sqrt{\frac{V_{max}}{T_{max}}}$$

\subsection*{Rayleigh-Ritz Method}

Assume displacement shape $x(y, t) = \phi(y)q(t)$

Use energy methods to find approximate $\omega_n$

\section*{Multi-Degree-of-Freedom (MDOF) Systems}

\subsection*{Equations of Motion (Matrix Form)}

$$\mathbf{M}\ddot{\mathbf{x}} + \mathbf{C}\dot{\mathbf{x}} + \mathbf{K}\mathbf{x} = \mathbf{f}(t)$$

where $\mathbf{M}$ is mass matrix, $\mathbf{C}$ is damping matrix, $\mathbf{K}$ is stiffness matrix

\subsection*{Free Vibration (Undamped)}

$$\mathbf{M}\ddot{\mathbf{x}} + \mathbf{K}\mathbf{x} = \mathbf{0}$$

\textbf{Assume harmonic solution:}
$$\mathbf{x}(t) = \boldsymbol{\phi}e^{j\omega t}$$

\textbf{Eigenvalue Problem:}
$$(\mathbf{K} - \omega^2\mathbf{M})\boldsymbol{\phi} = \mathbf{0}$$

\textbf{Characteristic Equation:}
$$\det(\mathbf{K} - \omega^2\mathbf{M}) = 0$$

Solving gives $n$ natural frequencies $\omega_1, \omega_2, \ldots, \omega_n$ and corresponding mode shapes $\boldsymbol{\phi}_1, \boldsymbol{\phi}_2, \ldots, \boldsymbol{\phi}_n$

\subsection*{Orthogonality of Mode Shapes}

$$\boldsymbol{\phi}_i^T\mathbf{M}\boldsymbol{\phi}_j = 0 \quad (i \neq j)$$
$$\boldsymbol{\phi}_i^T\mathbf{K}\boldsymbol{\phi}_j = 0 \quad (i \neq j)$$

\textbf{Normalized Mode Shapes:}
$$\boldsymbol{\phi}_i^T\mathbf{M}\boldsymbol{\phi}_i = 1$$
$$\boldsymbol{\phi}_i^T\mathbf{K}\boldsymbol{\phi}_i = \omega_i^2$$

\subsection*{Modal Analysis}

\textbf{Modal Coordinate Transformation:}
$$\mathbf{x}(t) = \boldsymbol{\Phi}\mathbf{q}(t) = \sum_{i=1}^{n} \boldsymbol{\phi}_i q_i(t)$$

where $\boldsymbol{\Phi} = [\boldsymbol{\phi}_1 \, \boldsymbol{\phi}_2 \, \cdots \, \boldsymbol{\phi}_n]$

\textbf{Decoupled Modal Equations:}

For proportional damping:
$$\ddot{q}_i + 2\zeta_i\omega_i\dot{q}_i + \omega_i^2 q_i = \frac{\boldsymbol{\phi}_i^T\mathbf{f}(t)}{\boldsymbol{\phi}_i^T\mathbf{M}\boldsymbol{\phi}_i}$$

Each modal equation is a SDOF system!

\section*{Vibration Absorbers}

\subsection*{Undamped Dynamic Absorber}

\textbf{Primary system with absorber attached:}

Primary mass $m_1$ with stiffness $k_1$

Absorber mass $m_2$ with stiffness $k_2$

\textbf{Tuning Condition (to eliminate vibration of primary mass):}
$$\omega_2 = \sqrt{\frac{k_2}{m_2}} = \omega$$

where $\omega$ is the forcing frequency

\textbf{Mass Ratio:}
$$\mu = \frac{m_2}{m_1}$$

\textbf{Effect:} Primary mass amplitude becomes zero at tuned frequency, but two new resonance peaks appear nearby

\subsection*{Damped Dynamic Absorber}

Adding damping $c_2$ to absorber provides:
\begin{itemize}
\item Reduces peak amplitudes at resonances
\item Primary mass never reaches exactly zero amplitude
\item Better performance over range of frequencies
\end{itemize}

\section*{Torsional Vibrations}

\textbf{Equation of Motion:}
$$I\ddot{\theta} + c_t\dot{\theta} + k_t\theta = M(t)$$

\textbf{Natural Frequency:}
$$\omega_n = \sqrt{\frac{k_t}{I}}$$

\textbf{Torsional Stiffness of Shaft:}
$$k_t = \frac{GJ}{L}$$

where $G$ is shear modulus, $J$ is polar moment of inertia, $L$ is length

For circular shaft: $J = \frac{\pi d^4}{32}$

\section*{Vibration Measurement and Testing}

\subsection*{Measurement Techniques}

\textbf{From Free Decay:}
\begin{itemize}
\item Natural frequency: $\omega_d = \frac{2\pi}{T_d}$
\item Damping ratio: Use logarithmic decrement $\zeta = \frac{\delta}{2\pi}$
\end{itemize}

\textbf{From Frequency Response:}
\begin{itemize}
\item Natural frequency: Peak of magnitude plot
\item Damping ratio: Half-power bandwidth method
\end{itemize}

\textbf{Half-Power Bandwidth Method:}

At points where $|H| = \frac{|H|_{max}}{\sqrt{2}}$:
$$\zeta = \frac{\omega_2 - \omega_1}{2\omega_n}$$

\subsection*{Fast Fourier Transform (FFT)}

Convert time domain signal to frequency domain to identify:
\begin{itemize}
\item Natural frequencies (peaks in FFT)
\item Dominant frequency components
\item Harmonic content
\end{itemize}

$$X(f) = \text{FFT}[x(t)]$$

\section*{Quick Reference Formulas}

\textbf{Natural Frequency:}
$$f_n = \frac{\omega_n}{2\pi} = \frac{1}{2\pi}\sqrt{\frac{k}{m}} \quad \text{(Hz)}$$

\textbf{Static Deflection Method:}
$$\omega_n = \sqrt{\frac{g}{\delta_{st}}}$$

where $\delta_{st}$ is static deflection under weight

\textbf{Amplitude at Resonance (underdamped):}
$$X_{max} \approx \frac{F_0}{2\zeta k}$$

\textbf{Transmissibility at $r = \sqrt{2}$:}
$$T = 1 \quad \text{(crossover point)}$$
